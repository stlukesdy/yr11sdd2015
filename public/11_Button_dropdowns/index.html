11 Button dropdowns

Problem

Read button dropdown selection.

Html

<select id="selectControl" class="form-control">
  <option value="one">One</option>
  <option value="two">Two</option>
  <option value="three">Three</option>
  <option value="four">Four</option>
  <option value="five">Five</option>
</select>

Javascript

$(document).ready(function () {
  $("#runButton").click(program);
  $('#selectControl').change(selectControlChanged);
});

var selectControlChanged = function () {
  console.log('changed');
};

var program = function () {
  var $select = $('#selectControl option:selected');
  var selectedValue = $select.val();
  var selectedText = $select.text();
  console.log('value = ' + selectedValue + ' text = ' + selectedText);
};

Open this document in Chrome and open the developer tools Windows: F12 or Ctrl-shift+I, Mac: Cmd + Opt + I.

Click run button.

Observe the console output.
