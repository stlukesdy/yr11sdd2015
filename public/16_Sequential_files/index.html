16 Sequential files

Creating a file

Sequential files need to be opened for output when they are created and must be closed before the program ends.

Printing the contents of a file

Sequential files that already exist need to be opened for input before they can be read, and closed before the program ends.

Problem

Create a file and write to it. Open the file and read from it.

Pseudocode

productArray(1).id = 1
productArray(1).name = "Oven cleaner kit"

productArray(2).id = 2
productArray(2).name = "Dishwasher tablets"

#
# Creating a file
#
Open ProductData for output

FOR i = 1 to 2 STEP 1
  Write ProductData from productArray(i).id, productArray(i).name
NEXT i

Write ProductData from "xxx", "xxx"

Close ProductData

#
# Printing the contents of a file
#
Open ProductData2 for input

Read id, name from ProductData2

WHILE id <> "xxx"
  Display id, name
  Read id, name from ProductData2
END WHILE

Close ProductData2

Javascript

$(document).ready(function () {
  $("#writeButton").click(writeProgram);
  $("#readButton").click(readProgram);
});

var writeProgram = function () {

  var productArray = [];

  productArray[0] = {};
  productArray[0].id = 1;
  productArray[0].name = 'Oven cleaner kit';

  productArray[1] = {};
  productArray[1].id = 2;
  productArray[1].name = 'Dishwasher tablets';

  var productData = [];

  for (var i = 0; i < 2; i = i + 1) {
    productData.push(productArray[i]);
  }

  localStorage.setItem('productData', JSON.stringify(productData));

};

var readProgram = function () {

  var producData2 = JSON.parse(localStorage.getItem('productData'));

  for (var i = 0; i < producData2.length; i = i + 1) {
    console.log(producData2[i].id + " " + producData2[i].name);
  }

};

Open this document in Chrome and open the developer tools Windows: F12 or Ctrl-shift+I, Mac: Cmd + Opt + I.

Click Write and Read buttons.

Observe the console output and the Resources --> Local Storage.
